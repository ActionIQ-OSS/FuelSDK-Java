203  protected static <T extends ETSoapObject> ETResponse<T> retrieve(ETClient client,
204  String soapObjectName,
205  ETFilter filter,
206  String continueRequest,
207  Class<T> type)
208  throws ETSdkException
209  {
210  ETResponse<T> response = new ETResponse<T>();
211 
212  //
213  // Get handle to the SOAP connection:
214  //
215 
216  ETSoapConnection connection = client.getSoapConnection();
217 
218  //
219  // Automatically refresh the token if necessary:
220  //
221 
222  client.refreshToken();
223 
224  //
225  // Read internal type from the SoapObject annotation:
226  //
227 
228  Class<T> externalType = type; // for code readability
229 
230  SoapObject internalTypeAnnotation
231  = externalType.getAnnotation(SoapObject.class);
232  assert internalTypeAnnotation != null;
233  Class<? extends APIObject> internalType = internalTypeAnnotation.internalType();
234  assert internalType != null;
235 
236  ETExpression expression = filter.getExpression();
237 
238  //
239  // Determine properties to retrieve:
240  //
241 
242  List<String> externalProperties = filter.getProperties();
243  List<String> internalProperties = null;
244 
245  if (externalProperties.size() > 0) {
246  //
247  // Only request those properties specified:
248  //
249 
250  internalProperties = new ArrayList<String>();
251 
252  for (String externalProperty : externalProperties) {
253  String internalProperty =
254  getInternalProperty(externalType, externalProperty);
255  assert internalProperty != null;
256  internalProperties.add(internalProperty);
257  }
258  } else {
259  //
260  // No properties were explicitly requested:
261  //
262 
263  internalProperties = getInternalProperties(externalType);
264 
265  //
266  // Remove properties that are unretrievable:
267  //
268 
269  for (String property : internalTypeAnnotation.unretrievable()) {
270  internalProperties.remove(property);
271  }
272  }
273 
274  //
275  // Perform the SOAP retrieve:
276  //
277 
278  Soap soap = connection.getSoap();
279 
280  RetrieveRequest retrieveRequest = new RetrieveRequest();
281 
282  if (continueRequest == null) {
283  // if soapObjectType is specified, use it; otherwise, default
284  // to the name of the internal class representing the object:
285  if (soapObjectName != null) {
286  retrieveRequest.setObjectType(soapObjectName);
287  } else {
288  retrieveRequest.setObjectType(internalType.getSimpleName());
289  }
290  retrieveRequest.getProperties().addAll(internalProperties);
291 
292  if (expression.getOperator() != null) {
293  //
294  // Convert the property names to their internal counterparts:
295  //
296 
297  String property = expression.getProperty();
298  if (property != null) {
299  expression.setProperty(getInternalProperty(type, property));
300  }
301  for (ETExpression subexpression : expression.getSubexpressions()) {
302  String p = subexpression.getProperty();
303  if (p != null) {
304  subexpression.setProperty(getInternalProperty(type, p));
305  }
306  }
307 
308  retrieveRequest.setFilter(toFilterPart(expression));
309  }
310  } else {
311  if (continueRequest != null) {
312  retrieveRequest.setContinueRequest(continueRequest);
313  }
314  }
315 
316  if (logger.isTraceEnabled()) {
317  logger.trace("RetrieveRequest:");
318  logger.trace(" objectType = " + retrieveRequest.getObjectType());
319  StringBuilder line = new StringBuilder(" properties = { ");
320  //String line = null;
321  for (String property : retrieveRequest.getProperties()) {
322 
323  line.append(property).append(", ");
324  }
325  if (retrieveRequest.getProperties().size() > 0) {
326  line.setLength(line.length() - 2);
327  /*if (line == null) {
328  line = " properties = { " + property;
329  } else {
330  line += ", " + property;
331  }*/
332  }
333  line.append(" }");
334  logger.trace(line.toString());
335  //logger.trace(line + " }");
336  if (filter != null) {
337  logger.trace(" filter = " + toFilterPart(expression));
338  }
339  }
340 
341  logger.trace("calling soap.retrieve...");
342 
343  RetrieveRequestMsg retrieveRequestMsg = new RetrieveRequestMsg();
344  retrieveRequestMsg.setRetrieveRequest(retrieveRequest);
345 
346  RetrieveResponseMsg retrieveResponseMsg = soap.retrieve(retrieveRequestMsg);
347 
348  if (logger.isTraceEnabled()) {
349  logger.trace("RetrieveResponseMsg:");
350  logger.trace(" requestId = " + retrieveResponseMsg.getRequestID());
351  logger.trace(" overallStatus = " + retrieveResponseMsg.getOverallStatus());
352  logger.trace(" results = {");
353  for (APIObject result : retrieveResponseMsg.getResults()) {
354  logger.trace(" " + result);
355  }
356  logger.trace(" }");
357  }
358 
359  response.setRequestId(retrieveResponseMsg.getRequestID());
360  if (retrieveResponseMsg.getOverallStatus().equals("OK")) {
361  response.setStatus(ETResult.Status.OK);
362  } else if (retrieveResponseMsg.getOverallStatus().equals("Error")) {
363  response.setStatus(ETResult.Status.ERROR);
364  }
365  response.setResponseCode(retrieveResponseMsg.getOverallStatus());
366  response.setResponseMessage(retrieveResponseMsg.getOverallStatus());
367  for (APIObject internalObject : retrieveResponseMsg.getResults()) {
368  //
369  // Allocate a new (external) object:
370  //
371 
372  T externalObject = null;
373  try {
374  externalObject = externalType.newInstance();
375  } catch (Exception ex) {
376  throw new ETSdkException("could not instantiate "
377  + externalType.getName(), ex);
378  }
379 
380  externalObject.setClient(client);
381 
382  //
383  // Convert from internal representation:
384  //
385 
386  externalObject.fromInternal(internalObject);
387 
388  //
389  // Add result to the list of results:
390  //
391 
392  ETResult<T> result = new ETResult<T>();
393  result.setObject(externalObject);
394  response.addResult(result);
395  }
396 
397  if (retrieveResponseMsg.getOverallStatus().equals("MoreDataAvailable")) {
398  response.setMoreResults(true);
399  }
400 
401  return response;
402  }

411  public static <T extends ETSoapObject> ETResponse<T> create(ETClient client,
412  List<T> objects)
413  throws ETSdkException
414  {
415  ETResponse<T> response = new ETResponse<T>();
416 
417  if (objects == null || objects.size() == 0) {
418  response.setStatus(ETResult.Status.OK);
419  return response;
420  }
421 
422  //
423  // Get handle to the SOAP connection:
424  //
425 
426  ETSoapConnection connection = client.getSoapConnection();
427 
428  //
429  // Automatically refresh the token if necessary:
430  //

433 
434  //
435  // Perform the SOAP create:
436  //
437 
438  Soap soap = connection.getSoap();
439 
440  CreateRequest createRequest = new CreateRequest();
441  createRequest.setOptions(new CreateOptions());
442  for (T object : objects) {
443  object.setClient(client);
444  createRequest.getObjects().add(object.toInternal());
445  }
446 
447  if (logger.isTraceEnabled()) {
448  logger.trace("CreateRequest:");
449  logger.trace(" objects = {");
450  for (APIObject object : createRequest.getObjects()) {
451  logger.trace(" " + object);
452  }
453  logger.trace(" }");
454  }
455 
456  logger.trace("calling soap.create...");
457 
458  CreateResponse createResponse = soap.create(createRequest);
459 
460  if (logger.isTraceEnabled()) {
461  logger.trace("CreateResponse:");
462  logger.trace(" requestId = " + createResponse.getRequestID());
463  logger.trace(" overallStatus = " + createResponse.getOverallStatus());
464  logger.trace(" results = {");
465  for (CreateResult result : createResponse.getResults()) {
466  logger.trace(" " + result);
467  }
468  logger.trace(" }");
469  }
470 
471  response.setRequestId(createResponse.getRequestID());
472  if (createResponse.getOverallStatus().equals("OK")) {
473  response.setStatus(ETResult.Status.OK);
474  } else if (createResponse.getOverallStatus().equals("Error")) {
475  response.setStatus(ETResult.Status.ERROR);
476  }
477  response.setResponseCode(createResponse.getOverallStatus());
478  response.setResponseMessage(createResponse.getOverallStatus());
479  for (CreateResult createResult : createResponse.getResults()) {
480  //
481  // Allocate a new (external) object:
482  //
483 
484  @SuppressWarnings("unchecked")
485  Class<T> externalType = (Class<T>) objects.get(0).getClass();
486 
487  T externalObject = null;
488  try {
489  externalObject = externalType.newInstance();
490  } catch (Exception ex) {
491  throw new ETSdkException("could not instantiate "
492  + externalType.getName(), ex);
493  }
494 
495  externalObject.setClient(client);
496 
497  //
498  // Convert from internal representation:
499  //
500 
501  // not all SOAP calls return the object though some do..
502  APIObject internalObject = createResult.getObject();
503  if (internalObject != null) {
504  externalObject.fromInternal(createResult.getObject());
505  } else {
506  // XXX populate fields from the object passed to the call?
507  externalObject.setId(Integer.toString(createResult.getNewID()));
508  }
509 
510  //
511  // Add result to the list of results:
512  //
513 
514  ETResult<T> result = new ETResult<T>();
515  if (createResult.getStatusCode().equals("OK")) {
516  result.setStatus(ETResult.Status.OK);
517  } else if (createResult.getStatusCode().equals("Error")) {
518  result.setStatus(ETResult.Status.ERROR);
519  }
520  result.setResponseCode(createResult.getStatusCode());
521  result.setResponseMessage(createResult.getStatusMessage());
522  result.setErrorCode(createResult.getErrorCode());
523  if (result.getResponseCode().equals("OK")) { // XXX?
524  result.setObject(externalObject);
525  }
526  response.addResult(result);
527  }
528 
529  return response;
530  }
