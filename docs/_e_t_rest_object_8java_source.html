331  private static <T extends ETRestObject> ETResponse<T> createUpdateDelete(ETClient client,
332  Method method,
333  List<T> objects)
334  throws ETSdkException
335  {
336  ETResponse<T> response = new ETResponse<T>();
337 
338  if (objects == null || objects.size() == 0) {
339  response.setStatus(ETResult.Status.OK);
340  return response;
341  }

343  ETRestConnection connection = client.getRestConnection();
344

347  //
348 
349  client.refreshToken();
350 
351  //
352  // Read call details from the RestObject annotation:
353  //
354 
355  RestObject annotations = objects.get(0).getClass().getAnnotation(RestObject.class);
356 
357  assert annotations != null;
358 
359  String path = annotations.path();
360  String primaryKey = annotations.primaryKey();
361  String collection = annotations.collection();
362  String totalCount = annotations.totalCount();

366  logger.trace("collection: " + collection);
367  logger.trace("totalCount: " + totalCount);
368

370  // There's currently no way to do this in bulk, so
371  // we walk through the list of objects and create,
372  // update, or delete them one at a time:
373  //
374 
375  Gson gson = client.getGson();
376 
377  for (T object : objects) {
378  switch (method) {
379  case POST:
380  logger.trace("POST " + path);
381  break;
382  case PATCH:
383  logger.trace("PATCH " + path + "/" + object.getId());
384  break;
385  case DELETE:
386  logger.trace("DELETE " + path + "/" + object.getId());
387  break;
388  default:
389  throw new ETSdkException("invalid method: " + method);
390  }
391 
392  String requestPayload = null;
393  if (method != DELETE) {
394  // no request payload for deletes
395  requestPayload = gson.toJson(object);
396  if (logger.isTraceEnabled()) {
397  JsonParser jsonParser = new JsonParser();
398  JsonObject jsonObject = jsonParser.parse(requestPayload).getAsJsonObject();
399  String jsonPrettyPrinted = gson.toJson(jsonObject);
400  for (String line : jsonPrettyPrinted.split("\\n")) {
401  logger.trace(line);
402  }
403  }
404  }
405 
406  Response r = null;
407  switch (method) {
408  case POST:
409  r = connection.post(path, requestPayload);
410  break;
411  case PATCH:
412  r = connection.patch(path + "/" + object.getId(), requestPayload);
413  break;
414  case DELETE:
415  r = connection.delete(path + "/" + object.getId());
416  break;
417  default:
418  throw new ETSdkException("invalid method: " + method);
419  }
420 
421  ETResult<T> result = new ETResult<T>();
422  result.setRequestId(r.getRequestId());
423  if (r.getResponseCode() >= 200 && r.getResponseCode() <= 299) {
424  result.setStatus(ETResult.Status.OK);
425  } else if (r.getResponseCode() >= 400 && r.getResponseCode() <= 599) {
426  result.setStatus(ETResult.Status.ERROR);
427  }
428  result.setResponseCode(r.getResponseCode().toString());
429  result.setResponseMessage(r.getResponseMessage());
430  if (method != DELETE) {
431  // no response payload for deletes
432  String responsePayload = r.getResponsePayload();
433  JsonParser jsonParser = new JsonParser();
434  JsonObject jsonObject = jsonParser.parse(responsePayload).getAsJsonObject();
435  if (logger.isTraceEnabled()) {
436  String jsonPrettyPrinted = gson.toJson(jsonObject);
437  for (String line : jsonPrettyPrinted.split("\\n")) {
438  logger.trace(line);
439  }
440  }
441  @SuppressWarnings("unchecked")
442  T responseObject = (T) gson.fromJson(responsePayload, object.getClass());
443  responseObject.setClient(client); // XXX
444  result.setObject(responseObject);
445  }
446 
447  response.addResult(result);
448 
449  object.setClient(client); // XXX
450  }
451 
452  // XXX set overall status
453 
454  return response;

513  logger.trace("totalCount = " + response.getTotalCount());
514

516  response.setMoreResults(true);

520 
521  for (JsonElement element : elements) {
522  ETResult<T> result = new ETResult<T>();
523  result.setObject(deserialize(client, element.toString(), type));
524  response.addResult(result);
525  }
526  } else {
527  ETResult<T> result = new ETResult<T>();
528  result.setObject(deserialize(client, jsonObject.toString(), type));
529  response.addResult(result);
530  }
531 
532  return response;

564  private static String toFilterString(String value, boolean forceQuotes)
565  throws ETSdkException
566  {
567  if (value.equals("")) {
568  forceQuotes = true;
569  }
570  // XXX investigate
571  if (value.contains("*")) {
572  forceQuotes = true;
573  }
574  // XXX workaround for FUEL-3348--remove after 02
575  if (value.contains("-")) {
576  forceQuotes = true;
577  }
578  boolean quotes = forceQuotes;
579  if (!forceQuotes) {
580  // needs quotes in the URL if there's whitespace
581  for (int i = 0; i < value.length(); i++) {
582  if (Character.isWhitespace(value.charAt(i))) {
583  quotes = true;
584  break;
585  }
586  }
587  }
588  String v = null;
589  if (quotes) {
590  v = "'" + value + "'";
591  } else {
592  v = value.toString();
593  }
594  try {
595  return URLEncoder.encode(v, "UTF-8");
596  } catch (UnsupportedEncodingException ex) {
597  throw new ETSdkException("error URL encoding " + v, ex);
598  }
599  }

620  private static String toFilterString(ETExpression expression, boolean first)
621  throws ETSdkException
622  {
623  StringBuilder stringBuilder = new StringBuilder();
624  if (first) {
625  stringBuilder.append("$filter=");
626  }
627  ETExpression.Operator operator = expression.getOperator();
628  if (operator == null) {
629  return null;
630  }
631  switch (operator) {
632  case AND:
633  stringBuilder.append(toFilterString(expression.getSubexpressions().get(0), false));
634  stringBuilder.append("%20");
635  stringBuilder.append("and");
636  stringBuilder.append("%20");
637  stringBuilder.append(toFilterString(expression.getSubexpressions().get(1), false));
638  break;
639  case OR:
640  stringBuilder.append(toFilterString(expression.getSubexpressions().get(0), false));
641  stringBuilder.append("%20");
642  stringBuilder.append("or");
643  stringBuilder.append("%20");
644  stringBuilder.append(toFilterString(expression.getSubexpressions().get(1), false));
645  break;
646  case NOT:
647  stringBuilder.append("not");
648  stringBuilder.append("%20");
649  stringBuilder.append(toFilterString(expression.getSubexpressions().get(0), false));
650  break;
651  case EQUALS:
652  stringBuilder.append(expression.getProperty());
653  stringBuilder.append("%20");
654  stringBuilder.append("eq");
655  stringBuilder.append("%20");
656  stringBuilder.append(toFilterString(expression.getValue(), false));
657  break;
658  case NOT_EQUALS:
659  stringBuilder.append(expression.getProperty());
660  stringBuilder.append("%20");
661  stringBuilder.append("neq");
662  stringBuilder.append("%20");
663  stringBuilder.append(toFilterString(expression.getValue(), false));
664  break;
665  case LESS_THAN:
666  stringBuilder.append(expression.getProperty());
667  stringBuilder.append("%20");
668  stringBuilder.append("lt");
669  stringBuilder.append("%20");
670  stringBuilder.append(toFilterString(expression.getValue(), false));
671  break;
672  case LESS_THAN_OR_EQUALS:
673  stringBuilder.append(expression.getProperty());
674  stringBuilder.append("%20");
675  stringBuilder.append("lte");
676  stringBuilder.append("%20");
677  stringBuilder.append(toFilterString(expression.getValue(), false));
678  break;
679  case GREATER_THAN:
680  stringBuilder.append(expression.getProperty());
681  stringBuilder.append("%20");
682  stringBuilder.append("gt");
683  stringBuilder.append("%20");
684  stringBuilder.append(toFilterString(expression.getValue(), false));
685  break;
686  case GREATER_THAN_OR_EQUALS:
687  stringBuilder.append(expression.getProperty());
688  stringBuilder.append("%20");
689  stringBuilder.append("gte");
690  stringBuilder.append("%20");
691  stringBuilder.append(toFilterString(expression.getValue(), false));
692  break;
693  case IS_NULL:
694  stringBuilder.append(expression.getProperty());
695  stringBuilder.append("%20");
696  stringBuilder.append("is");
697  stringBuilder.append("%20");
698  stringBuilder.append("null");
699  break;
700  case IS_NOT_NULL:
701  stringBuilder.append(expression.getProperty());
702  stringBuilder.append("%20");
703  stringBuilder.append("is");
704  stringBuilder.append("%20");
705  stringBuilder.append("not");
706  stringBuilder.append("%20");
707  stringBuilder.append("null");
708  break;
709  case IN:
710  stringBuilder.append(expression.getProperty());
711  stringBuilder.append("%20");
712  stringBuilder.append("in");
713  stringBuilder.append("%20");
714  stringBuilder.append("(");
715  boolean firstValue = true;
716  for (String value : expression.getValues()) {
717  if (firstValue) {
718  firstValue = false;
719  } else {
720  stringBuilder.append(",");
721  }
722  stringBuilder.append(toFilterString(value));
723  }
724  stringBuilder.append(")");
725  break;
726  case BETWEEN:
727  stringBuilder.append(expression.getProperty());
728  stringBuilder.append("%20");
729  stringBuilder.append("between");
730  stringBuilder.append("%20");
731  stringBuilder.append(toFilterString(expression.getValues().get(0), false));
732  stringBuilder.append("%20");
733  stringBuilder.append("and");
734  stringBuilder.append("%20");
735  stringBuilder.append(toFilterString(expression.getValues().get(1), false));
736  break;
737  case LIKE:
738  stringBuilder.append(expression.getProperty());
739  stringBuilder.append("%20");
740  stringBuilder.append("like");
741  stringBuilder.append("%20");
742  stringBuilder.append(toFilterString(expression.getValue(), true));
743  break;
744  default:
745  throw new ETSdkException("unsupported operator: " + operator);
746  }
747 
748  return stringBuilder.toString();
749  }
