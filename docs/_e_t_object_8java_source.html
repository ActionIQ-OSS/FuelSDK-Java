92  // @ExternalName implies @PrettyPrint

94  field.getAnnotation(ExternalName.class);
95 
96  if (prettyPrintAnnotation == null && externalNameAnnotation == null) {
97  continue;
98  }
99 
100  String name = null;
101  if (externalNameAnnotation != null) {
102  name = externalNameAnnotation.value();
103  } else {
104  name = field.getName();
105  }
106  String value = null;
107  try {
108  // briefly change accessibility so we can get value
109  boolean isAccessible = field.isAccessible();
110  if (!isAccessible) {
111  field.setAccessible(true);
112  }
113  if (field.get(this) != null) {
114  value = field.get(this).toString();
115  }
116  field.setAccessible(isAccessible);
117  } catch (IllegalAccessException ex) {
118  throw new AssertionError("should never ever get here");
119  }
120  if (value == null) {
121  continue;
122  }
123  if (toStringMultiLine) {
124  for (int i = 0; i < currentIndentLevel; i++) {
125  stringBuilder.append(" ");
126  }
127  } else {
128  if (first) {
129  first = false;
130  } else {
131  stringBuilder.append(",");
132  }
133  }
134  if (toStringSpaceAroundEquals) {
135  stringBuilder.append(name + " = " + value);
136  } else {
137  stringBuilder.append(name + "=" + value);
138  }
139  if (toStringMultiLine) {
140  stringBuilder.append(System.getProperty("line.separator"));
141  }
142  }
143 
144  currentIndentLevel -= toStringMultiLineIndentAmount;
145 
146  if (toStringMultiLine) {
147  for (int i = 0; i < currentIndentLevel; i++) {
148  stringBuilder.append(" ");
149  }
150  }

153
