302  public synchronized String requestToken(String refreshToken)
303  throws ETSdkException
304  {
305  if (clientId == null || clientSecret == null) {
306  // no-op
307  return null;
308  }
309 
310  logger.debug("requesting access token...");
311 
312  //
313  // Construct the JSON payload. Set accessType to offline so
314  // we get a refresh token. Pass in current refresh token if
315  // we have one:
316  //
317 
318  JsonObject jsonObject = new JsonObject();
319  jsonObject.addProperty("clientId", clientId);
320  jsonObject.addProperty("clientSecret", clientSecret);
321  jsonObject.addProperty("accessType", (configuration.get("accessType") != null ? configuration.get("accessType") :"online"));
322  if (configuration.get("accessType") != null && configuration.get("accessType").equals("offline") && refreshToken != null){
323  jsonObject.addProperty("refreshToken", refreshToken);
324  }
325 
326  String requestPayload = gson.toJson(jsonObject);
327 
328  ETRestConnection.Response response = null;
329  if (configuration.isTrue("requestLegacyToken")) {
330  response = authConnection.post(PATH_REQUESTTOKEN_LEGACY, requestPayload);
331  } else {
332  response = authConnection.post(PATH_REQUESTTOKEN, requestPayload);
333  }
334 
335  if (response.getResponseCode() != HttpURLConnection.HTTP_OK) {
336  throw new ETSdkException("error obtaining access token "
337  + "("
338  + response.getResponseCode()
339  + " "
340  + response.getResponseMessage()
341  + ")");
342  }
343 
344  //
345  // Parse the JSON response into the appropriate instance
346  // variables:
347  //
348 
349  String responsePayload = response.getResponsePayload();
350 
351  JsonParser jsonParser = new JsonParser();
352  jsonObject = jsonParser.parse(responsePayload).getAsJsonObject();
353  logger.debug("received token:");
354  this.accessToken = jsonObject.get("accessToken").getAsString();
355  logger.debug(" accessToken: " + this.accessToken);
356  this.expiresIn = jsonObject.get("expiresIn").getAsInt();
357  logger.debug(" expiresIn: " + this.expiresIn);
358  JsonElement jsonElement = jsonObject.get("legacyToken");
359  if (jsonElement != null) {
360  this.legacyToken = jsonElement.getAsString();
361  }
362  logger.debug(" legacyToken: " + this.legacyToken);
363  if (jsonObject.get("refreshToken") != null){
364  this.refreshToken = jsonObject.get("refreshToken").getAsString();
365  }
366 
367  logger.debug(" refreshToken: " + this.refreshToken);
368 
369  //
370  // Calculate the token expiration time. As before,
371  // System.currentTimeMills() is in milliseconds so
372  // we multiply expiresIn by 1000:
373  //
374 
375  tokenExpirationTime = System.currentTimeMillis() + (expiresIn * 1000);
376 
377  logger.debug("access token expires at " + new Date(tokenExpirationTime));
378 
379  return accessToken;
380  }

387  public synchronized String refreshToken()
388  throws ETSdkException
389  {
390  if (tokenExpirationTime > 0) {
391  logger.debug("access token expires at " + new Date(tokenExpirationTime));
392 
393  //
394  // If the current token expires more than five
395  // minutes from now, we don't need to refresh
396  // (tokenExpirationTime and System.currentTimeMills()
397  // are in milliseconds so we multiply by 1000):
398  //
399 
400  if (tokenExpirationTime - System.currentTimeMillis() > 5*60*1000) {
401  logger.debug("not refreshing access token");
402  return accessToken;
403  }
404 
405  logger.debug("refreshing access token...");
406 
407  if (refreshToken == null) {
408  throw new ETSdkException("refreshToken == null");
409  }
410  }
411 
412  requestToken(refreshToken);
413 
414  soapConnection.setAccessToken(accessToken);
415 
416  return accessToken;
417  }
