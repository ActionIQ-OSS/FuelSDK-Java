124  public static ETFilter parse(String... s)
125  throws ETSdkException
126  {
127  ETFilter filter = new ETFilter();
128 
129  for (String t : s) {
130  try {
131  filter.setExpression(ETExpression.parse(t));
132  } catch (ETSdkException ex) {
133  if (ex.getCause() instanceof ParseException) {
134  //
135  // It's not an expression, so it's either
136  // an order by clause or it's a property:
137  //
138 
139  if (t.length() >= 8
140  && t.substring(0, 8).toLowerCase().equals("order by")) {
141  //
142  // Order by clause:
143  //
144 
145  //t = t.toLowerCase();
146  String tokens[] = t.substring(9).split(" ");
147 
148  if (tokens.length > 1
149  && tokens[1].toLowerCase().equals("desc")) {
150  filter.setOrderByAsc(false);
151  }
152 
153  filter.setOrderBy(Arrays.asList(tokens[0].split(",")));
154  } else {
155  //
156  // Property:
157  //
158 
159  filter.addProperty(t);
160  }
161  } else {
162  throw ex;
163  }
164  }
165  }
166 
167  return filter;
168  }
